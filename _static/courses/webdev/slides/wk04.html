Complex Lookups

Field Lookups แบบขั้นสูง

# ค้นหาข้อมูลด้วยเงื่อนไขซับซ้อน
          Book.objects.filter(title__icontains="python")     # Contains (case-insensitive)
          Book.objects.filter(price__range=(100, 500))       # Between values
          Book.objects.filter(published_date__year=2023)     # Extract year
          Book.objects.filter(author__name__startswith="J")  # Related field lookup

Date/Time Lookups

from datetime import datetime, timedelta

          # ค้นหาหนังสือที่ออกในเดือนนี้
          this_month = datetime.now().month
          Book.objects.filter(published_date__month=this_month)

          # ค้นหาหนังสือที่ออกใน 30 วันที่ผ่านมา
          last_30_days = datetime.now() - timedelta(days=30)
          Book.objects.filter(published_date__gte=last_30_days)

Q Objects และ Complex Queries

การใช้ Q Objects สำหรับเงื่อนไขซับซ้อน

from django.db.models import Q

          # OR condition
          Book.objects.filter(
          Q(category__name="Fiction") | Q(category__name="Romance")
          )

          # AND + OR combination
          Book.objects.filter(
          Q(price__lt=300) & (Q(category__name="Fiction") | Q(author__name="John"))
          )

          # NOT condition
          Book.objects.filter(~Q(category__name="Horror"))

💡 Tips:

  • & = AND operation
  • | = OR operation
  • ~ = NOT operation
  • ใช้วงเล็บ () เพื่อจัดกลุมเงื่อนไข

Aggregation และ Annotation

การคำนวณข้อมูลในฐานข้อมูล

from django.db.models import Count, Avg, Sum, Max, Min

          # นับจำนวนหนังสือของแต่ละผู้แต่ง
          Author.objects.annotate(book_count=Count('book'))

          # หาราคาเฉลี่ยของหนังสือแต่ละหมวดหมู่
          Category.objects.annotate(avg_price=Avg('book__price'))

          # รวมยอดขายทั้งหมด
          Book.objects.aggregate(total_value=Sum('price'))

⚠️ ความแตกต่าง:

  • aggregate(): คำนวณค่าเดียวสำหรับทั้ง QuerySet
  • annotate(): เพิ่มคอลัมน์คำนวณให้แต่ละ object

Prefetch และ Select Related

ป้องกัน N+1 Query Problem

# select_related สำหรับ ForeignKey และ OneToOne
          books = Book.objects.select_related('author', 'category').all()

          # prefetch_related สำหรับ ManyToMany และ Reverse ForeignKey
          authors = Author.objects.prefetch_related('book_set').all()

          # Combined usage
          books = Book.objects.select_related('author')\
          .prefetch_related('author__book_set').all()

✅ select_related

  • ใช้ SQL JOIN
  • ForeignKey, OneToOne
  • 1 query เดียว

✅ prefetch_related

  • ใช้ separate queries
  • ManyToMany, Reverse FK
  • 2+ queries แต่มีประสิทธิภาพ

Advanced Filtering Techniques

Subqueries และ Exists

from django.db.models import Exists, OuterRef

          # ค้นหา Author ที่มีหนังสือราคาแพงกว่า 500
          expensive_books = Book.objects.filter(price__gt=500, author=OuterRef('pk'))
          authors_with_expensive_books = Author.objects.filter(
          Exists(expensive_books)
          )

Subquery สำหรับการเปรียบเทียบ

from django.db.models import Subquery

          newest_book_per_author = Book.objects.filter(
          author=OuterRef('author')
          ).order_by('-published_date').values('published_date')[:1]

          books = Book.objects.annotate(
          latest_date=Subquery(newest_book_per_author)
          )
